```javascript
// Full category management system
class CategoryManager {
    constructor() {
        this.categories = {
            income: [],
            expense: [],
            transfer: [],
            investment: [],
            debt: [],
            savings: [],
            asset: []
        };
    }
    
    getCategoryType(type) {
        switch(type) {
            case 'income': return 'Income';
            case 'expense': return 'Expense';
            case 'transfer': return 'Transfer';
            case 'investment': return 'Investment';
            case 'debt': return 'Debt';
            case 'savings': return 'Savings';
            case 'asset': return 'Asset';
            default: return 'Other';
        }
    }
    
    getCategoryColor(type) {
        const colors = {
            'income': '#10b981',
            'expense': '#ef4444',
            'transfer': '#6b7280',
            'investment': '#3b82f6',
            'debt': '#dc2626',
            'savings': '#8b5cf6',
            'asset': '#f59e0b'
        };
        return colors[type] || '#6b7280';
    }
}
```

## **7. Net Worth Calculation:**

With these categories, you can calculate real net worth:
```javascript
function calculateNetWorth() {
    const totalAssets = transactions
        .filter(t => t.type === 'asset' || t.type === 'investment')
        .reduce((sum, t) => sum + (t.assetValue || t.amount), 0);
    
    const totalSavings = transactions
        .filter(t => t.type === 'savings')
        .reduce((sum, t) => sum + t.amount, 0);
    
    const totalDebt = transactions
        .filter(t => t.type === 'debt')
        .reduce((sum, t) => sum + t.amount, 0);
    
    return totalAssets + totalSavings - totalDebt;
}
```
